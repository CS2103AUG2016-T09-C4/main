	public java.util.function.Predicate<? super Task> getTasksWithSameNameOrTags(Set<String> args) {
		return t -> {
			try {
				return t.getName().contains(args) || t.getTags().contains(args);
			} catch (IllegalValueException e) {
				return false;
			}
		};
	}

		};

		}
